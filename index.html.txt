    const arena = document.getElementById('arena');
    const noBtn = document.getElementById('no');
    const yesBtn = document.getElementById('yes');
    const result = document.getElementById('result');

   
    const dodge = () => {
      const pad = 12;

      const arenaRect = arena.getBoundingClientRect();
      const btnRect = noBtn.getBoundingClientRect();

      const maxX = arenaRect.width - btnRect.width - pad;
      const maxY = arenaRect.height - btnRect.height - pad;

      const x = Math.max(pad, Math.random() * maxX);
      const y = Math.max(pad, Math.random() * maxY);

      noBtn.style.left = x + "px";
      noBtn.style.top  = y + "px";

      spawnHeart(arenaRect.width, arenaRect.height);
    };

    noBtn.addEventListener('mouseenter', dodge);
    noBtn.addEventListener('touchstart', (e) => {
      e.preventDefault(); // —á—Ç–æ–±—ã –Ω–µ "–Ω–∞–∂–∞–ª–æ—Å—å"
      dodge();
    }, { passive: false });

    yesBtn.addEventListener('click', () => {
      result.classList.add('show');
      confettiHearts();
      arena.style.display = "none";
    });


    function spawnHeart(w, h){
      const heart = document.createElement('div');
      heart.className = 'heart';
      heart.textContent = Math.random() > 0.5 ? 'üíó' : 'üíû';
      heart.style.left = (Math.random() * (w - 20)) + "px";
      heart.style.top = (h - 30) + "px";
      heart.style.fontSize = (14 + Math.random()*18) + "px";
      arena.appendChild(heart);
      setTimeout(() => heart.remove(), 3500);
    }

        h.className='heart';
        h.textContent = ['üíñ','üíò','üíù','üíó'][Math.floor(Math.random()*4)];

        h.style.fontSize = (16 + Math.random()*26) + "px";
        document.body.appendChild(h);
        setTimeout(()=>h.remove(), 3500);
      }
    }
